      accept: :json,
      content_type: :xml
  }

def delete(url, options={}, &block)
  http_options = options.merge(@basic_options)
  if block_given?
    RestClient.delete(url, http_options, &block)
  else
    RestClient.delete(url, http_options)
  end
end
