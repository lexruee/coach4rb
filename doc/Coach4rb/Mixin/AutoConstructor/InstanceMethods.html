def [](key)
  if value = instance_variable_get("@#{key}")
    value
  else
    send key rescue nil
  end
end

def initialize(a_hash={})
  raise 'Param a_hash is not a hash!' unless a_hash.is_a? Hash
  a_hash.each do |key, value|
    instance_variable_set("@#{key}", value)
  end
end
